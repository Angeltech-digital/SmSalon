# SOLUTION FOR SSL ERROR
# If pip SSL fails, use this workaround:

# Option 1: Use HTTP instead of HTTPS
# Run this command:
pip install -r requirements.txt \
  --trusted-host pypi.python.org \
  --trusted-host pypi.org \
  --trusted-host files.pythonhosted.org \
  -i http://pypi.python.org/simple/

# Option 2: Install packages individually
pip install Django==4.2.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install djangorestframework==3.14.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install django-cors-headers==4.0.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install python-decouple==3.8 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install celery==5.3.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install redis==4.5.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install Pillow==9.5.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org
pip install requests==2.31.0 --trusted-host pypi.org --trusted-host files.pythonhosted.org

# Option 3: Install from local/offline (if packages downloaded)
pip install --no-index --find-links=/path/to/packages -r requirements.txt
